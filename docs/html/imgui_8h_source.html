25 //
26 // [L:16-1][N:16-1]
27 // XOR
28 // [F:32 --------1]
29 //
30 // Generate a *fake* unique ID for gui controls at compiling/run time
31 //
32 #define GenUIID(N) ( ((__LINE__<<16) | ( N & 0xFFFF))^((uintptr_t)&__FILE__) )
33 //
34 // GenUIID(0) will give a unique ID at each source code line.
35 // If you need one UI ID per line, just call GenUIID with 0
36 //
37 // GenUIID(0)
38 //
39 // But, two GenUIID(0) calls at the same line will give the same ID.
40 //
41 // So, in a while/for loop body, GenUIID(0) will give you the same ID.
42 // In this case, you need call GenUIID with different N parameter:
43 //
44 // GenUIID(N)
45 //
46 //===========================================================================
